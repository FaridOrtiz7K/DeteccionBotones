# PyAutoGUI para control del mouse y captura de pantalla
pip install pyautogui

# OpenCV para procesamiento de imágenes
pip install opencv-python

# NumPy para operaciones matemáticas (generalmente se instala con OpenCV)
pip install numpy

# Pillow para manipulación de imágenes
pip install Pillow

# Si estás en Windows, instala también pygetwindow para manejar ventanas
pip install pygetwindow

#Para el uso de las teclas de la computadora
pip install keyboard opencv-python pyautogui numpy

#Se usa para detectar las pantallas y se encuentre la pagina q queremos ejecutar el programa
pip install psutil
